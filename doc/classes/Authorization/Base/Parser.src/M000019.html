      def parse_role_of_model( str )
        role_regex = '^\s*(\'\s*(.+)\s*\'|(\w+))\s+'
        model_regex = '\s+(:*\w+)$'
        parse_regex = Regexp.new(role_regex + '(' + VALID_PREPOSITIONS + ')' + model_regex)
        if str =~ parse_regex
          role = $2 || $3
          model = $5          
          model_obj = get_model( model )
          yield role, model_obj if block_given?
          return true
        end
        false
      end
