      def evaluate_role( str )
        logger.info("evaluate_model(#{str})")
        parse_role( str ) do |role|
          logger.info("  in block: role(#{role})")
          begin
            if @current_user.nil?
              @stack.push( false )
            else
              @stack.push( @current_user.has_role?( role ) )
            end
          rescue
            @stack.push( false )
          end
        end
      end
