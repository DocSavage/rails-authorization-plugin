      def evaluate_role_of_model( str )
        logger.info("evaluate_role_of_model(#{str})")
        parse_role_of_model( str ) do |role, model| 
          logger.info("  in block: role(#{role}), model(#{model})")

          # Make sure we have a model that implements necessary #accepts_role?
          if not model.methods.include? "accepts_role?"
            raise( ModelDoesntImplementRoles, "Model (#{model}) doesn't implement #accepts_role?" )
            return false
          end

          # Query the model
          has_permission = model.send( :accepts_role?, role, @current_user )
          logger.info("     model -> #{@current_user.inspect}")
          logger.info("     has_permission -> #{has_permission.inspect}")
          @stack.push( has_permission )
        end
      end
