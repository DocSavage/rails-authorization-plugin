      def parse_or( str )
        if str =~ /^\s*(.+)\s+or\s+(.+)$/ 
          can_parse = parse_term( $1 ) and parse_expr( $2 )
          @stack.push( @stack.pop | @stack.pop ) if can_parse
        else
          can_parse = false
        end
        can_parse
      end
