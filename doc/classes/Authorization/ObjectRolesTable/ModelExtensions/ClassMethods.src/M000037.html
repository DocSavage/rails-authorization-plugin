        def acts_as_authorizable
          has_many :roles, :as => :authorizable
          has_many :authorizable_users, :through => :roles, :class_name => 'User'
          
          def accepts_role?( role_name, user )
            role_covers_all_objects = user.roles.find_by_name_and_authorizable_type_and_authorizable_id( role_name, nil, nil)
            if not role_covers_all_objects
              role_covers_class = user.roles.find_by_name_and_authorizable_type_and_authorizable_id( role_name, self.to_s, nil)
            end
            ( role_covers_all_objects or role_covers_class ) ? true : false
          end
          
          def accepts_role( role_name, user )
            role = Role.find_by_name_and_authorizable_id_and_authorizable_type( role_name, self.to_s, nil )
            role = Role.create( :name => role_name, :authorizable_type => self.to_s ) if role.nil?
            user.roles << role if not user.roles.include?( role )
          end
          
          def accepts_no_role( role_name, user )
            role = Role.find_by_name_and_authorizable_id_and_authorizable_type( role_name, self.to_s, nil )
            user.roles.delete( role ) if role
          end
          
          include Authorization::ObjectRolesTable::ModelExtensions::InstanceMethods
        end
