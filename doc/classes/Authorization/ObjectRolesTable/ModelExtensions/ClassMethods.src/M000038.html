          def accepts_role?( role_name, user )
            role_covers_all_objects = user.roles.find_by_name_and_authorizable_type_and_authorizable_id( role_name, nil, nil)
            if not role_covers_all_objects
              role_covers_class = user.roles.find_by_name_and_authorizable_type_and_authorizable_id( role_name, self.to_s, nil)
            end
            ( role_covers_all_objects or role_covers_class ) ? true : false
          end
