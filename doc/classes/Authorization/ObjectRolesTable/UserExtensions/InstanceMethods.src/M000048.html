        def has_no_role( role_name, authorizable_object = nil  )
          if authorizable_object
            role = authorizable_object.roles.find_by_name( role_name )
          else
            role = self.roles.find_by_name_and_authorizable_id( role_name, nil )
          end
          self.roles.delete( role ) if role
        end
