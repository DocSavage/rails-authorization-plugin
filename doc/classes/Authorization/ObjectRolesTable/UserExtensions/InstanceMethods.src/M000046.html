        def has_role?( role_name, authorizable_object = nil )
          if authorizable_object
            role = self.roles.find_by_name_and_authorizable_type_and_authorizable_id( role_name, authorizable_object.class.to_s, authorizable_object.id )
          else
            role = self.roles.find_by_name( role_name )
          end
          role ? true : false
        end
